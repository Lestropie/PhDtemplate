%=================== REDEFINE DEFAULT FIGURE PLACEMENT ========================
\makeatletter
% This influences float placement; letters are in order of priority. Here it has
% been set to prefer placing figures & tables at the top of the page ('t'), or 
% place multiple floats onto a single page ('p') if necessary
% Default is {tbp}, where 'b' is bottom of page
\def\fps@figure{tp}
% With the default table of contents and lists of tables/figures, I found that
% sometimes the text would run over where the numbers appear, with the numbers
% correspondingly forced over the page margin. These commands give the numbers
% a bit more space and seems to solve the problem.
\renewcommand{\@pnumwidth}{3em} 
\renewcommand{\@tocrmarg}{4em}
